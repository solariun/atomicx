205 bool atomicx::SelectNextThread()
206 {
207 atomicx* pItem = ms_paFirst;
208
209 do
210 {
211 if (ms_pCurrent->m_nTargetTime == 0 && pItem->m_nTargetTime > 0)
212 {
213 ms_pCurrent = pItem;
214 }
215
216 if (pItem->m_aStatus == aTypes::start || pItem->m_aStatus == aTypes::now)
217 {
218 ms_pCurrent = pItem;
219 break;
220 }
221 else if ((pItem->m_aStatus == aTypes::sleep || (pItem->m_aStatus == aTypes::wait && pItem->m_nTargetTime > 0)) && ms_pCurrent->m_nTargetTime >= pItem->m_nTargetTime)
222 {
223 ms_pCurrent = pItem;
224 }
225
226 } while ((pItem = pItem->m_paNext));
227
228 if (ms_pCurrent == nullptr)
229 {
230 return false;
231 }
232 else
233 {
234 switch (ms_pCurrent->m_aStatus)
235 {
236 case aTypes::wait:
237 if (ms_pCurrent->m_nTargetTime > 0)
238 {
239 ms_pCurrent->m_aSubStatus = aSubTypes::timeout;
240 }
241 else
242 {
243 // A blocked wait should never come here if the system was not in deadlock.
244 return false;
245 }
246
247 case aTypes::sleep:
248 {
249 atomicx_time nCurrent = Atomicx_GetTick();
250
251 (void) Atomicx_SleepTick(nCurrent < ms_pCurrent->m_nTargetTime ? ms_pCurrent->m_nTargetTime - nCurrent : 0);
252
253 break;
254 }
255
256 case aTypes::running:
257 case aTypes::start:
258 case aTypes::now:
259 break;
260
261 default:
262 return false;
263 }
264 }
265
266 return true;
267 }

308 bool atomicx::Yield(atomicx_time nSleep)
309 {
310 m_LastUserExecTime = GetCurrentTick () - m_lastResumeUserTime;
311
312 if (m_aStatus == aTypes::running)
313 {
314 m_aStatus = aTypes::sleep;
315 m_aSubStatus = aSubTypes::ok;
316 m_nTargetTime=Atomicx_GetTick() + (nSleep == ATOMICX_TIME_MAX ? m_nice : nSleep);
317 }
318 else if (m_aStatus == aTypes::wait)
319 {
320 m_nTargetTime=nSleep > 0 ? nSleep + Atomicx_GetTick() : 0;
321 }
322 else
323 {
324 m_nTargetTime = (atomicx_time)~0;
325 }
326
327 volatile uint8_t nStackEnd=0;
328 m_pStaskEnd = &nStackEnd;
329 m_stacUsedkSize = (size_t) (m_pStaskStart - m_pStaskEnd);
330
331 if (m_stacUsedkSize > m_stackSize || m_stack == nullptr)
332 {
333 /*
334 * Controll the auto-stack memory
335 * Note: Due to some small microcontroller
336 * does not have try/catch/throw by default
337 * I decicde to use malloc/free instead
338 * to control errors
339 */
340
341 if (m_flags.autoStack == true)
342 {
343 if (m_stack != nullptr)
344 {
345 free ((void*) m_stack);
346 }
347
348 if (m_stacUsedkSize > m_stackSize)
349 {
350 m_stackSize = m_stacUsedkSize + m_stackIncreasePace;
351 }
352
353 if ((m_stack = (uint8_t*) malloc (m_stacUsedkSize)) == nullptr)
354 {
355 m_aStatus = aTypes::stackOverflow;
356 }
357 }
358 else
359 {
360 m_aStatus = aTypes::stackOverflow;
361 }
362
363 if (m_aStatus == aTypes::stackOverflow)
364 {
365 (void) StackOverflowHandler();
366 abort();
367 }
368 }
369
370 if (m_aStatus != aTypes::stackOverflow && memcpy((void*)m_stack, (const void*) m_pStaskEnd, m_stacUsedkSize) != (void*) m_stack)
371 {
372 return false;
373 }
374
375 if (setjmp(m_context) == 0)
376 {
377 longjmp(ms_joinContext, 1);
378 }
379 else
380 {
381 ms_pCurrent->m_stacUsedkSize = (size_t) (ms_pCurrent->m_pStaskStart - &nStackEnd);
382 if (memcpy((void*) ms_pCurrent->m_pStaskEnd, (const void*) ms_pCurrent->m_stack, ms_pCurrent->m_stacUsedkSize) != (void*) ms_pCurrent->m_pStaskEnd)
383 {
384 return false;
385 }
386 }
387
388 ms_pCurrent->m_lastResumeUserTime = GetCurrentTick ();
389
390 ms_pCurrent->m_aStatus = aTypes::running;
391
392 if (ms_pCurrent->m_flags.dynamicNice == true)
393 {
394 ms_pCurrent->m_nice = ((ms_pCurrent->m_LastUserExecTime) + ms_pCurrent->m_nice) / 2;
395 }
396
397 return true;
398 }

639 uint16_t atomicx::crc16(const uint8_t* pData, size_t nSize, uint16_t nCRC)
640 {
641 #define POLY 0x8408
642 unsigned char nCount;
643 unsigned int data;
644
645 nCRC = ~nCRC;
646
647 if (nSize == 0) return (~nCRC);
648
649 do
650 {
651 for (nCount = 0, data = (unsigned int)0xff & *pData++; nCount < 8; nCount++, data >>= 1)
652 {
653 if ((nCRC & 0x0001) ^ (data & 0x0001))
654 nCRC = (nCRC >> 1) ^ POLY;
655 else
656 nCRC >>= 1;
657 }
658 } while (--nSize);
659
660 nCRC = ~nCRC;
661 data = nCRC;
662 nCRC = (nCRC << 8) | (data >> 8 & 0xff);
663
664 return (nCRC);
665 }
